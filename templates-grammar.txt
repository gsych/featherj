template =
	import-directive*
	members-directive*
	(code-block | text)*

import-directive =
	T_OPEN IMPORT text T_CLOSE

members-directive =
	T_OPEN MEMBERS text T_CLOSE

code-block =
	(T_OPEN | T_ECHO) text T_CLOSE

text =
	line (LT line?)*

line =
	( DQUOTE | SLASH | TEXT_SPAN )+

T_OPEN = '<%'
T_ECHO = '<%='
T_CLOSE = '%>'
DQUOTE = '"'
SLASH = '\'
IMPORT = 'import'
MEMBERS = 'members'
TEXT_SPAN = (.)* ~(T_OPEN | T_ECHO | T_CLOSE | LT | DQUOTE | SLASH)
