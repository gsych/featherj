template =
	import-directive*
	constructor-directive*
	(code-block | text)*

import-directive =
	T_OPEN IMPORT code-block T_CLOSE

constructor-directive =
	T_OPEN CONSTRUCTOR code-block T_CLOSE

code-block =
	(T_OPEN | T_ECHO) text T_CLOSE

text =
	line (LT line?)*

line =
	( DQUOTE | SLASH | TEXT_SPAN )+

T_OPEN = '<%'
T_ECHO = '<%='
T_CLOSE = '%>'
DQUOTE = '"'
SLASH = '\'
IMPORT = 'import'
CONSTRUCTOR = 'constructor'
TEXT_SPAN = (.)* ~(T_OPEN | T_ECHO | T_CLOSE | LT | DQUOTE | SLASH)
